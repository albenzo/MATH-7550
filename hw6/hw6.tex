\begin{problem}
  On the 2-sphere, consider the flow
  \[
    \theta(t,\langle x,y,z\rangle)
    =\langle x,y\cos(t)-z\sin(t),y\sin(t)+z\cos(t)\rangle
  \]
  Find the vector field on $S^2$ induced by this flow.
\end{problem}
